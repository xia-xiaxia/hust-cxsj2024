// 测试语法错误的代码示例
// 这个文件包含各种语法错误

int main() {
    // 缺少分号
    int x = 10
    int y = 20;  // 错误：上一行缺少分号
    
    // 缺少表达式
    int z =;  // 错误：赋值语句缺少右侧表达式
    
    // 括号不匹配
    if (x > 5 {  // 错误：缺少右括号
        y = y + 1;
    }
    
    // 缺少大括号
    while (y > 0)
        x = x - 1;
        y = y - 1;  // 错误：while循环体缺少大括号
    }  // 错误：多余的右大括号
    
    // 语法结构错误
    else {  // 错误：else没有对应的if
        return 1;
    }
    
    // 缺少返回语句的分号
    return 0  // 错误：缺少分号
